ULFM OpenMPI install
----------------------
cd /home/username/
wget https://download.open-mpi.org/release/open-mpi/v5.0/openmpi-5.0.0rc9.tar.gz
mkdir ULFM
mv openmpi-5.0.0rc9.tar.gz ULFM/
cd ULFM/
tar -xzf openmpi-5.0.0rc9.tar.gz 
cd openmpi-5.0.0rc9/
mkdir build
cd build

// ../configure --prefix=/home/disfavour/ULFM --with-ft=ulfm  - не сработал
../configure --prefix=/home/disfavour/ULFM --with-ft=ulfm --with-pmix=internal

make -j4
make install
cd ../..
ls bin/
